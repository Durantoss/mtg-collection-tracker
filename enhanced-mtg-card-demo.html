Integration Example

// Basic usage
const cardData = {
    id: 1,
    name: "Lightning Bolt",
    manaCost: "{R}",
    type: "Instant",
    set: "Alpha",
    setCode: "LEA",
    rarity: "common",
    condition: "NM",
    quantity: 4,
    isFoil: false,
    imageUrl: "https://example.com/card.jpg",
    prices: {
        scryfall: { price: 2.50, lastUpdated: "2024-01-01" },
        tcgplayer: { price: 2.75, lastUpdated: "2024-01-01" }
    },
    notes: "Great condition, tournament legal"
};

// Create card component
const cardComponent = new MTGCardComponent(cardData, {
    enableQuantityControls: true,
    enablePriceTracking: true,
    enableDeckBuilder: true,
    enableSwipeActions: true
});

// Add to DOM
document.getElementById('collection').appendChild(cardComponent.getElement());

// Listen for events
cardComponent.getElement().addEventListener('mtg-card-update', (event) => {
    const { type, cardData } = event.detail;
    
    switch (type) {
        case 'quantity-changed':
            updateCollectionStats();
            break;
        case 'add-to-deck':
            addCardToDeck(cardData);
            break;
        case 'remove-card':
            removeFromCollection(cardData.id);
            break;
    }
});
